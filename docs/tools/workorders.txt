# Prompts Service - Work Queue Management
  class workorders(Service):  # Service for managing prompt files in a work queue system.
    def __init__(self, server_instance)
    def urgency(self, *args) -> str  # Get or set default urgency level for new workorders.
    def _sanitize_filename(self, description)  # Convert description to safe filename component.
    def _format_prompt_list(self, dir_name, files)  # Format a numbered list of workorders with urgency and tags.
    def add(self, *args) -> str  # Create a new workorder with metadata and execution mode support.
    def list(self, *args) -> str  # List workorders in specified directory or all directories.
    def read(self, filename: str) -> str  # Read and display a prompt file.
    def edit(self, *args) -> str  # Edit an existing prompt file.
    def append(self, *args) -> str  # Append content to an existing prompt file.
    def _resolve_prompt(self, ref, from_dir='ready')  # Resolve a prompt reference (number or filename) to a filepath.
    def assign(self, *args) -> str  # Assign a workorder to an agent with execution mode support.
    def move(self, *args) -> str  # Move a prompt between directories.
    def delete(self, filename: str) -> str  # Delete a prompt file.
    def status(self) -> str  # Show status of all prompt directories.
    def archive(self, filename: str) -> str  # Archive a completed prompt from done/ to archive/.
    def default(self, *args) -> str  # Show available commands.
  class prompts(workorders):  # Compatibility alias for legacy prompts service.
