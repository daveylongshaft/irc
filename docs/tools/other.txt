    def _detect_geography(self)  # Detect timezone and locale-based geography.
    def _detect_time(self)  # Check system time accuracy.
    def _detect_network(self)  # Detect hostname and IP addresses.
    def _run_version_cmd(self, binary, args=None)  # Run a command and return version string or None.
    def _detect_software(self)  # Detect installed software tools and their versions.
    def _detect_docker(self)  # Detect Docker availability, daemon status, and resources.
    def _detect_ai_agents(self)  # Detect which AI CLI tools are available in PATH.
    def _assess_resources(self)  # Assess whether this box can handle Docker + AI workloads.
    def _detect_runtime(self)  # Detect runtime directories (temp root, agent work base) in both Windows and Linux notations.
    def agent_temp_root(self)  # Get system temp directory path.
    def agent_work_base(self)  # Get CSC agent work base directory path.
    def run_dir(self)  # Get the runtime state and log directory path.
    def _get_platform_json_path(self)  # Get the path for platform.json (in project root).
    def _persist_platform(self)  # Write platform data to platform.json atomically.
    def _install_background_services(self)  # Install background service scripts (queue-worker, test-runner) when install-at-startup flag is active.
    def _setup_windows_services(self)  # Setup Windows Task Scheduler for background services.
    def _setup_unix_services(self)  # Setup Linux/macOS cron entries for background services.
    def _setup_android_services(self)  # Setup Android/Termux services (requires manual cron or SystemD user service).
    def load_platform_json(path=None)  # Load platform.json from disk. Static method for use by services.
    def has_tool(self, tool_name)  # Check if a software tool is installed.
    def has_docker(self)  # Check if Docker is usable.
    def matches_platform(self, required_platforms)  # Check if this system matches one of the required platforms.
    def has_min_ram(self, min_ram_str)  # Check if system meets minimum RAM requirement.
    def check_requirements(self, requires=None, platform_list=None, min_ram=None)  # Check if this system satisfies all requirements.
    def refresh_platform(self)  # Re-run all detection and persist.
    def get_abs_root_path(self, components)  # Return absolute path from PROJECT_ROOT.
    def get_abs_tmp_path(self, components)  # Return absolute path from TMP directory (PROJECT_ROOT/tmp).
    def store_path(self, name, path_type, components)  # Store a path reference in platform.json for later retrieval.
    def load_path(self, name)  # Retrieve a stored path reference.
    def export_env_paths(self)  # Export critical paths as environment variables.
  def _parse_size(size_str)  # Parse a human-readable size string to bytes.
